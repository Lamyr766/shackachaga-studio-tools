Studio Tools — One-time setup required. Enter your Supabase credentials to connect.
Step 1 — Go to supabase.com, create free project named shackachaga
Step 2 — Run the SQL script below in Supabase SQL Editor
Step 3 — Go to Project Settings → API, copy URL and anon key below
SQL Setup Script:
CREATE TABLE IF NOT EXISTS projects ( id bigint primary key generated always as identity, name text, piece text, email text, phone text, city text, source text, dims text, wood text, style text, vision text, budget text, timeline text, deadline text, amount numeric default 0, balance numeric default 0, status text default 'new', review text default 'No', delivery text, notes text, created text, updated_at timestamptz default now() ); CREATE TABLE IF NOT EXISTS wood_stock ( id bigint primary key generated always as identity, species text, grade text, quantity numeric, unit text, cost_per_unit numeric, supplier text, notes text, updated_at timestamptz default now() ); CREATE TABLE IF NOT EXISTS saved_posts ( id bigint primary key generated always as identity, product text, tone text, platform text, en_caption text, en_hashtags text, fr_caption text, fr_hashtags text, created_at timestamptz default now() ); ALTER TABLE projects ENABLE ROW LEVEL SECURITY; ALTER TABLE wood_stock ENABLE ROW LEVEL SECURITY; ALTER TABLE saved_posts ENABLE ROW LEVEL SECURITY; CREATE POLICY "auth_all" ON projects FOR ALL USING (auth.role() = 'authenticated') WITH CHECK (auth.role() = 'authenticated'); CREATE POLICY "auth_all" ON wood_stock FOR ALL USING (auth.role() = 'authenticated') WITH CHECK (auth.role() = 'authenticated'); CREATE POLICY "auth_all" ON saved_posts FOR ALL USING (auth.role() = 'authenticated') WITH CHECK (auth.role() = 'authenticated'); CREATE TABLE IF NOT EXISTS messages ( id bigint primary key generated always as identity, sender_email text not null, sender_name text, content text not null, room text default 'team', created_at timestamptz default now() ); CREATE TABLE IF NOT EXISTS webrtc_signals ( id bigint primary key generated always as identity, from_user text, to_user text, signal_type text, payload jsonb, created_at timestamptz default now() ); CREATE TABLE IF NOT EXISTS user_locations ( id bigint primary key generated always as identity, user_email text unique, display_name text, lat numeric, lng numeric, status text default 'offline', location_type text default 'shop', last_seen timestamptz default now() ); ALTER TABLE messages ENABLE ROW LEVEL SECURITY; ALTER TABLE webrtc_signals ENABLE ROW LEVEL SECURITY; ALTER TABLE user_locations ENABLE ROW LEVEL SECURITY; DO $$ BEGIN IF NOT EXISTS (SELECT 1 FROM pg_policies WHERE tablename='messages' AND policyname='auth_all') THEN CREATE POLICY "auth_all" ON messages FOR ALL USING (auth.role()='authenticated') WITH CHECK (auth.role()='authenticated'); END IF; IF NOT EXISTS (SELECT 1 FROM pg_policies WHERE tablename='webrtc_signals' AND policyname='auth_all') THEN CREATE POLICY "auth_all" ON webrtc_signals FOR ALL USING (auth.role()='authenticated') WITH CHECK (auth.role()='authenticated'); END IF; IF NOT EXISTS (SELECT 1 FROM pg_policies WHERE tablename='user_locations' AND policyname='auth_all') THEN CREATE POLICY "auth_all" ON user_locations FOR ALL USING (auth.role()='authenticated') WITH CHECK (auth.role()='authenticated'); END IF; END $$; CREATE TABLE IF NOT EXISTS clients ( id bigint primary key generated always as identity, first_name text, last_name text, email text, phone text, city text, source text, preferred_wood text, notes text, total_spent numeric default 0, project_count int default 0, created_at timestamptz default now(), updated_at timestamptz default now() ); CREATE TABLE IF NOT EXISTS project_notes ( id bigint primary key generated always as identity, project_id bigint references projects(id) on delete cascade, author text, note text, type text default 'note', created_at timestamptz default now() ); CREATE TABLE IF NOT EXISTS time_logs ( id bigint primary key generated always as identity, user_email text, project_id bigint, clock_in timestamptz, clock_out timestamptz, duration_minutes numeric, location_type text default 'shop', lat numeric, lng numeric, notes text, created_at timestamptz default now() ); CREATE TABLE IF NOT EXISTS expenses ( id bigint primary key generated always as identity, project_id bigint, description text, amount numeric default 0, category text, date text, created_at timestamptz default now() ); ALTER TABLE clients ENABLE ROW LEVEL SECURITY; ALTER TABLE project_notes ENABLE ROW LEVEL SECURITY; ALTER TABLE time_logs ENABLE ROW LEVEL SECURITY; ALTER TABLE expenses ENABLE ROW LEVEL SECURITY; CREATE POLICY IF NOT EXISTS "auth_all" ON clients FOR ALL USING (auth.role() = 'authenticated') WITH CHECK (auth.role() = 'authenticated'); CREATE POLICY IF NOT EXISTS "auth_all" ON project_notes FOR ALL USING (auth.role() = 'authenticated') WITH CHECK (auth.role() = 'authenticated'); CREATE POLICY IF NOT EXISTS "auth_all" ON time_logs FOR ALL USING (auth.role() = 'authenticated') WITH CHECK (auth.role() = 'authenticated'); CREATE POLICY IF NOT EXISTS "auth_all" ON expenses FOR ALL USING (auth.role() = 'authenticated') WITH CHECK (auth.role() = 'authenticated');
